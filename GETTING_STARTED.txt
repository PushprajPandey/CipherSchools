💡 SAMPLE QUERIES
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Assignment 1: SELECT * FROM customers;

Assignment 2: SELECT * FROM customers WHERE city = 'New York';

Assignment 3: SELECT c.name, COUNT(o.id) as order_count
              FROM customers c
              INNER JOIN orders o ON c.id = o.customer_id
              GROUP BY c.name;

Assignment 4: SELECT c.name, SUM(o.total) as total_spent
              FROM customers c
              INNER JOIN orders o ON c.id = o.customer_id
              GROUP BY c.name
              ORDER BY total_spent DESC
              LIMIT 5;
